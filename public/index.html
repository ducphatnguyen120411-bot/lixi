async function handleLixi() { const name = document.getElementById('user-name').value.trim(); const money = Math.random() > 0.5 ? "20.000đ" : "10.000đ"; // Tỉ lệ random const res = await fetch('/api/lixi', { method: 'POST', headers: {'Content-Type': 'application/json'}, body: JSON.stringify({ name, amount: money }) }); const result = await res.json(); if (result.status === 'exists') { alert("Bạn đã nhận rồi: " + result.data.amount); } else { alert("Chúc mừng! Bạn nhận được: " + result.data.amount); } }
