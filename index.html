const express = require('express'); const mysql = require('mysql2/promise'); const path = require('path'); const app = express(); app.use(express.json()); app.use(express.static('public')); // Kết nối MySQL bằng biến môi trường của Railway const pool = mysql.createPool(process.env.MYSQL_URL || process.env.MYSQLURL); // Khởi tạo bảng dữ liệu nếu chưa có async function initDB() { const conn = await pool.getConnection(); await conn.query(` CREATE TABLE IF NOT EXISTS users ( id VARCHAR(255) PRIMARY KEY, name VARCHAR(255), amount VARCHAR(50), time DATETIME DEFAULT CURRENT_TIMESTAMP ) `); conn.release(); } initDB(); // API xử lý Lì Xì app.post('/api/lixi', async (req, res) => { const { name, amount } = req.body; const id = name.toLowerCase().replace(/\s+/g, ''); try { const [rows] = await pool.query('SELECT * FROM users WHERE id = ?', [id]); if (rows.length > 0) return res.json({ status: 'exists', data: rows[0] }); await pool.query('INSERT INTO users (id, name, amount) VALUES (?, ?, ?)', [id, name, amount]); res.json({ status: 'success', data: { name, amount } }); } catch (err) { res.status(500).send(err.message); } }); // API lấy lịch sử app.get('/api/history', async (req, res) => { const [rows] = await pool.query('SELECT name, amount FROM users ORDER BY time DESC LIMIT 10'); res.json(rows); }); app.listen(process.env.PORT || 3000);
